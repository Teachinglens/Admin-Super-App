var SPREADSHEET_ID = "1nHhl5HQVHRwh4XTT7uY05bocGlgOMB1TjLZ3ScFbNFI";
var SHEET_USERS = "USERS";
var SHEET_DATA = "DATA";

function getSS() {
  try {
    return SpreadsheetApp.getActiveSpreadsheet() || SpreadsheetApp.openById(SPREADSHEET_ID);
  } catch (e) {
    return SpreadsheetApp.openById(SPREADSHEET_ID);
  }
}

function setup() {
  var ss = getSS();
  
  // Setup Users Sheet
  if (!ss.getSheetByName(SHEET_USERS)) {
    var userSheet = ss.insertSheet(SHEET_USERS);
    userSheet.appendRow(["Timestamp", "Email", "Password", "Package", "DownloadsLeft", "IsActivated", "ActivationCode", "NamaGelar", "Role"]);
  }
  
  // Setup Data Sheet
  var dataSheet = ss.getSheetByName(SHEET_DATA);
  if (!dataSheet) {
    dataSheet = ss.insertSheet(SHEET_DATA);
    dataSheet.appendRow(["Timestamp", "Email", "Nama Guru", "Mapel", "Materi", "Jenis Dokumen", "Kelas", "Semester", "Tahun"]);
  } else {
    // Ensure headers are correct if sheet exists
    dataSheet.getRange(1, 1, 1, 9).setValues([["Timestamp", "Email", "Nama Guru", "Mapel", "Materi", "Jenis Dokumen", "Kelas", "Semester", "Tahun"]]);
  }
}

function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  var action = data.action;
  
  if (action === "register") return registerUser(data);
  if (action === "login") return loginUser(data);
  if (action === "activate") return activateUser(data);
  if (action === "trackDownload") return trackDownload(data);
  if (action === "getUsers") return getUsers(data);
  if (action === "updateUser") return updateUser(data);
  if (action === "getUser") return getUser(data);
  if (action === "saveData") return saveData(data);
  
  return response({ status: "error", message: "Action not found" });
}

function registerUser(data) {
  var ss = getSS();
  var sheet = ss.getSheetByName(SHEET_USERS);
  var rows = sheet.getDataRange().getValues();
  
  // Check if user exists
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][1] === data.email) {
      return response({ status: "error", message: "Email sudah terdaftar." });
    }
  }
  
  var activationCode = "GURU-" + Math.random().toString(36).substr(2, 6).toUpperCase();
  var downloads = 30; // Default Basic
  var role = (data.email === "barlimahardikasandy@gmail.com") ? "admin" : "user";
  
  sheet.appendRow([
    new Date(),
    data.email,
    "", // Password will be set after activation
    "Basic", // Default package
    downloads,
    false,
    activationCode,
    data.namaGelar || "",
    role
  ]);
  
  return response({ status: "success", message: "Registrasi berhasil.", code: activationCode });
}

function activateUser(data) {
  var ss = getSS();
  var sheet = ss.getSheetByName(SHEET_USERS);
  var rows = sheet.getDataRange().getValues();
  
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][1] === data.email && rows[i][6] === data.code) {
      var password = Math.random().toString(36).substr(2, 8);
      sheet.getRange(i + 1, 3).setValue(password);
      sheet.getRange(i + 1, 6).setValue(true);
      
      var user = {
        email: rows[i][1],
        package: rows[i][3],
        downloadsLeft: rows[i][4],
        isActivated: true,
        password: password,
        namaGelar: rows[i][7],
        role: rows[i][8]
      };
      
      return response({ status: "success", user: user });
    }
  }
  
  return response({ status: "error", message: "Kode aktivasi salah." });
}

function loginUser(data) {
  var ss = getSS();
  var sheet = ss.getSheetByName(SHEET_USERS);
  var rows = sheet.getDataRange().getValues();
  
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][1] === data.email && rows[i][2] === data.password) {
      if (!rows[i][5]) return response({ status: "error", message: "Akun belum diaktivasi." });
      
      var user = {
        email: rows[i][1],
        package: rows[i][3],
        downloadsLeft: rows[i][4],
        isActivated: true,
        namaGelar: rows[i][7],
        role: rows[i][8]
      };
      
      return response({ status: "success", user: user });
    }
  }
  
  return response({ status: "error", message: "Email atau Password salah." });
}

function trackDownload(data) {
  var ss = getSS();
  var sheet = ss.getSheetByName(SHEET_USERS);
  var rows = sheet.getDataRange().getValues();
  
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][1] === data.email) {
      var current = rows[i][4];
      if (rows[i][3] === "Basic") {
        var newVal = Math.max(0, current - 1);
        sheet.getRange(i + 1, 5).setValue(newVal);
        return response({ status: "success", downloadsLeft: newVal });
      }
      return response({ status: "success", downloadsLeft: current });
    }
  }
  return response({ status: "error" });
}

function getUsers(data) {
  // Simple admin check
  if (data.adminEmail !== "barlimahardikasandy@gmail.com") return response({ status: "error", message: "Unauthorized" });
  
  var ss = getSS();
  var sheet = ss.getSheetByName(SHEET_USERS);
  var rows = sheet.getDataRange().getValues();
  var users = [];
  
  for (var i = 1; i < rows.length; i++) {
    users.push({
      email: rows[i][1],
      package: rows[i][3],
      downloadsLeft: rows[i][4],
      isActivated: rows[i][5],
      namaGelar: rows[i][7],
      role: rows[i][8]
    });
  }
  
  return response({ status: "success", users: users });
}

function updateUser(data) {
  if (data.adminEmail !== "barlimahardikasandy@gmail.com") return response({ status: "error", message: "Unauthorized" });
  
  var ss = getSS();
  var sheet = ss.getSheetByName(SHEET_USERS);
  var rows = sheet.getDataRange().getValues();
  
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][1] === data.targetEmail) {
      if (data.newPackage) sheet.getRange(i + 1, 4).setValue(data.newPackage);
      if (data.newDownloads !== undefined) sheet.getRange(i + 1, 5).setValue(data.newDownloads);
      return response({ status: "success" });
    }
  }
  return response({ status: "error" });
}

function getUser(data) {
  var ss = getSS();
  var sheet = ss.getSheetByName(SHEET_USERS);
  var rows = sheet.getDataRange().getValues();
  
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][1] === data.email) {
      return response({
        status: "success",
        user: {
          email: rows[i][1],
          package: rows[i][3],
          downloadsLeft: rows[i][4],
          isActivated: rows[i][5],
          namaGelar: rows[i][7],
          role: rows[i][8]
        }
      });
    }
  }
  return response({ status: "error", message: "User not found" });
}

function saveData(data) {
  var ss = getSS();
  var sheet = ss.getSheetByName(SHEET_DATA);
  if (!sheet) {
    setup();
    sheet = ss.getSheetByName(SHEET_DATA);
  }
  sheet.appendRow([
    new Date(),
    data.email,
    data.nama_guru,
    data.mapel,
    data.materi || "",
    data.jenis_dokumen,
    data.kelas || "",
    data.semester || "",
    data.tahun || ""
  ]);
  return response({ status: "success" });
}

function response(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}
